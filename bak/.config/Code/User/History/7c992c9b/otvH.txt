get_property(dialect_libs GLOBAL PROPERTY MLIR_DIALECT_LIBS)
get_property(conversion_libs GLOBAL PROPERTY MLIR_CONVERSION_LIBS)

set (LIBS
    ${dialect_libs}
    ${conversion_libs}
    AffineDistributeToMPI
    MLIROptLib
    MLIRPass
)

add_llvm_executable(topoly-opt topoly-opt.cpp)

llvm_update_compile_flags(topoly-opt)
target_link_libraries(topoly-opt PRIVATE ${LIBS})

mlir_check_all_link_libraries(topoly-opt)
install(TARGETS topoly-opt)
